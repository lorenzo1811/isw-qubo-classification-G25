"""
Streamlit GUI for the QUBO-based binary classification pipeline.

Run with:
    streamlit run src/qubo_project/gui.py
"""

import os
import json
import traceback

import pandas as pd
import streamlit as st

from .preprocessing import fit_normalize
from .feature_selection import select_features
from .model import train, predict

OUTPUT_DIR = "outputs"
CLASSIFIER_OPTIONS = ["random_forest", "logistic_regression", "gradient_boosting"]

os.makedirs(OUTPUT_DIR, exist_ok=True)


def _out_path(filename: str) -> str:
    """Build a relative output path inside the outputs/ folder."""
    return os.path.join(OUTPUT_DIR, filename)


def _read_header(csv_path: str):
    """Read only the header row of a CSV file to validate column names."""
    try:
        return list(pd.read_csv(csv_path, nrows=0).columns)
    except Exception:
        return []


def _init_state():
    defaults = {
        "input_csv": None,
        "target_column": "",
        "normalized_csv": None,
        "preproc_json": None,
        "reduced_train_csv": None,
        "reduced_test_csv": None,
        "optim_csv": None,
        "feat_sel_json": None,
        "model_path": None,
        "train_metrics_json": None,
        "predictions_csv": None,
        "classif_stats_json": None,
    }
    for key, value in defaults.items():
        if key not in st.session_state:
            st.session_state[key] = value


def _show_error(exc: Exception):
    st.error(f"An error occurred: {exc}")
    with st.expander("Show technical details"):
        st.code(traceback.format_exc())


def preprocessing_tab():
    st.header("1. Preprocessing")

    uploaded_path = st.text_input(
        "Path to input CSV dataset",
        value=st.session_state["input_csv"] or "",
        help="Relative path to the dataset file, e.g. data/input_dataset.csv",
    )
    target_column = st.text_input(
        "Target column name",
        value=st.session_state["target_column"],
    )
    min_perc_valid = st.slider(
        "Minimum percentage of valid (non-zero/non-null) data per column",
        min_value=0.01, max_value=0.99, value=0.05, step=0.01,
    )

    if st.button("Run preprocessing"):
        errors = []
        if not uploaded_path or not os.path.isfile(uploaded_path):
            errors.append("The input dataset path does not exist.")
        if not target_column.strip():
            errors.append("Target column name cannot be empty.")
        elif uploaded_path and os.path.isfile(uploaded_path):
            header = _read_header(uploaded_path)
            if target_column not in header:
                errors.append(
                    f"Target column '{target_column}' was not found in the dataset header."
                )
        if not (0 < min_perc_valid < 1):
            errors.append("minPercValid must be between 0 and 1.")

        if errors:
            for e in errors:
                st.error(e)
            return

        normalized_csv = _out_path("normalized.csv")
        preproc_json = _out_path("preprocessing_result.json")

        try:
            with st.spinner("Running preprocessing..."):
                fit_normalize(
                    input_csv=uploaded_path,
                    target_column=target_column,
                    normalized_csv=normalized_csv,
                    outInitalRes_json=preproc_json,
                    minPercValid=min_perc_valid,
                )
            st.session_state["input_csv"] = uploaded_path
            st.session_state["target_column"] = target_column
            st.session_state["normalized_csv"] = normalized_csv
            st.session_state["preproc_json"] = preproc_json
            st.success("Preprocessing completed.")
        except Exception as exc:
            _show_error(exc)
            return

    if st.session_state["preproc_json"] and os.path.isfile(st.session_state["preproc_json"]):
        st.subheader("Preprocessing statistics")
        with open(st.session_state["preproc_json"]) as f:
            st.json(json.load(f))


def feature_selection_tab():
    st.header("2. Feature Selection (QUBO)")

    if not st.session_state["normalized_csv"]:
        st.info("Run preprocessing first.")
        return

    st.write(f"Using normalized dataset: `{st.session_state['normalized_csv']}`")

    col1, col2 = st.columns(2)
    with col1:
        perc_test = st.slider("Test set percentage", 0.05, 0.95, 0.30, 0.05)
        perc_selected = st.slider("Percentage of features to select", 0.01, 0.99, 0.20, 0.01)
        allowance = st.number_input("Allowance (feature count tolerance)", min_value=0, value=1)
    with col2:
        seed = st.number_input("Random seed", min_value=0, value=42)
        alpha_computations = st.number_input(
            "Max number of alpha optimization steps", min_value=1, value=10
        )

    if st.button("Run feature selection"):
        errors = []
        if not (0 < perc_test < 1):
            errors.append("percTest must be between 0 and 1.")
        if not (0 < perc_selected < 1):
            errors.append("percSelected must be between 0 and 1.")
        if allowance < 0:
            errors.append("Allowance must be non-negative.")
        if alpha_computations < 1:
            errors.append("alpha_computations must be at least 1.")

        if errors:
            for e in errors:
                st.error(e)
            return

        reduced_train_csv = _out_path("training_reduced.csv")
        reduced_test_csv = _out_path("test_reduced.csv")
        optim_csv = _out_path("optimizations.csv")
        feat_sel_json = _out_path("feature_selection_result.json")

        try:
            with st.spinner("Running QUBO feature selection (this may take a while)..."):
                select_features(
                    normalized_csv=st.session_state["normalized_csv"],
                    reducedTrain_csv=reduced_train_csv,
                    reducedTest_csv=reduced_test_csv,
                    output_ottim_csv=optim_csv,
                    output_json=feat_sel_json,
                    target_column=st.session_state["target_column"],
                    percTest=perc_test,
                    percSelected=perc_selected,
                    allowance=int(allowance),
                    seed=int(seed),
                    alpha_computations=int(alpha_computations),
                )
            st.session_state["reduced_train_csv"] = reduced_train_csv
            st.session_state["reduced_test_csv"] = reduced_test_csv
            st.session_state["optim_csv"] = optim_csv
            st.session_state["feat_sel_json"] = feat_sel_json
            st.success("Feature selection completed.")
        except Exception as exc:
            _show_error(exc)
            return

    if st.session_state["feat_sel_json"] and os.path.isfile(st.session_state["feat_sel_json"]):
        st.subheader("Feature selection statistics")
        with open(st.session_state["feat_sel_json"]) as f:
            result = json.load(f)
        st.json(result)
        if "selected_feature_names" in result:
            st.write("Selected features:", result["selected_feature_names"])


def training_tab():
    st.header("3. Training")

    if not st.session_state["reduced_train_csv"]:
        st.info("Run feature selection first.")
        return

    st.write(f"Using reduced training dataset: `{st.session_state['reduced_train_csv']}`")

    classifier = st.selectbox("Classifier", CLASSIFIER_OPTIONS)
    seed = st.number_input("Random seed", min_value=0, value=42, key="train_seed")

    if st.button("Run training"):
        model_path = _out_path("model.joblib")
        metrics_json = _out_path("training_metrics.json")

        try:
            with st.spinner("Training classifier..."):
                train(
                    classifier=classifier,
                    reducedTrain_csv=st.session_state["reduced_train_csv"],
                    target_column=st.session_state["target_column"],
                    model_path=model_path,
                    metrics_json=metrics_json,
                    seed=int(seed),
                )
            st.session_state["model_path"] = model_path
            st.session_state["train_metrics_json"] = metrics_json
            st.success("Training completed.")
        except Exception as exc:
            _show_error(exc)
            return

    if st.session_state["train_metrics_json"] and os.path.isfile(st.session_state["train_metrics_json"]):
        st.subheader("Training metrics")
        with open(st.session_state["train_metrics_json"]) as f:
            st.json(json.load(f))


def prediction_tab():
    st.header("4. Prediction")

    default_test = st.session_state["reduced_test_csv"] or ""
    test_csv = st.text_input(
        "Path to reduced test dataset CSV",
        value=default_test,
        help="Defaults to the reduced test set from feature selection, but any compatible "
             "reduced CSV can be used.",
    )

    if not st.session_state["model_path"]:
        st.info("Run training first, or provide a model path manually below.")
        model_path = st.text_input("Path to trained model (.joblib)", value="")
    else:
        model_path = st.session_state["model_path"]
        st.write(f"Using trained model: `{model_path}`")

    if st.button("Run prediction"):
        errors = []
        if not test_csv or not os.path.isfile(test_csv):
            errors.append("The test dataset path does not exist.")
        if not model_path or not os.path.isfile(model_path):
            errors.append("The model path does not exist.")

        if errors:
            for e in errors:
                st.error(e)
            return

        predictions_csv = _out_path("predictions.csv")
        classif_stats_json = _out_path("classification_stats.json")

        try:
            with st.spinner("Running predictions..."):
                predict(
                    reduced_Test_csv=test_csv,
                    target_column=st.session_state["target_column"],
                    model_path=model_path,
                    predictions_csv=predictions_csv,
                    classif_stats_json=classif_stats_json,
                )
            st.session_state["predictions_csv"] = predictions_csv
            st.session_state["classif_stats_json"] = classif_stats_json
            st.success("Prediction completed.")
        except Exception as exc:
            _show_error(exc)
            return

    if st.session_state["classif_stats_json"] and os.path.isfile(st.session_state["classif_stats_json"]):
        st.subheader("Classification statistics")
        with open(st.session_state["classif_stats_json"]) as f:
            stats = json.load(f)
        st.json(stats)

        if "confusion_matrix" in stats:
            cm = stats["confusion_matrix"]["matrix"]
            labels = stats["confusion_matrix"]["labels"]
            cm_df = pd.DataFrame(cm, index=[f"actual_{l}" for l in labels],
                                  columns=[f"pred_{l}" for l in labels])
            st.subheader("Confusion matrix")
            st.dataframe(cm_df)

    if st.session_state["predictions_csv"] and os.path.isfile(st.session_state["predictions_csv"]):
        st.subheader("Predictions preview")
        preds_df = pd.read_csv(st.session_state["predictions_csv"])
        st.dataframe(preds_df.head(50))


def outputs_tab():
    st.header("5. Outputs")
    st.write("Files produced so far, available for download:")

    files = {
        "Normalized dataset": st.session_state["normalized_csv"],
        "Preprocessing stats (JSON)": st.session_state["preproc_json"],
        "Reduced training set": st.session_state["reduced_train_csv"],
        "Reduced test set": st.session_state["reduced_test_csv"],
        "Alpha optimizations (CSV)": st.session_state["optim_csv"],
        "Feature selection stats (JSON)": st.session_state["feat_sel_json"],
        "Trained model": st.session_state["model_path"],
        "Training metrics (JSON)": st.session_state["train_metrics_json"],
        "Predictions (CSV)": st.session_state["predictions_csv"],
        "Classification stats (JSON)": st.session_state["classif_stats_json"],
    }

    any_file = False
    for label, path in files.items():
        if path and os.path.isfile(path):
            any_file = True
            with open(path, "rb") as f:
                st.download_button(
                    label=f"Download {label} ({os.path.basename(path)})",
                    data=f.read(),
                    file_name=os.path.basename(path),
                    key=path,
                )
    if not any_file:
        st.info("No output files generated yet.")


def main():
    st.set_page_config(page_title="QUBO Classification Pipeline", layout="wide")
    st.title("QUBO-based Binary Classification Pipeline")
    _init_state()

    tab_names = ["Preprocessing", "Feature Selection", "Training", "Prediction", "Outputs"]
    tabs = st.tabs(tab_names)

    with tabs[0]:
        preprocessing_tab()
    with tabs[1]:
        feature_selection_tab()
    with tabs[2]:
        training_tab()
    with tabs[3]:
        prediction_tab()
    with tabs[4]:
        outputs_tab()


if __name__ == "__main__":
    main()
